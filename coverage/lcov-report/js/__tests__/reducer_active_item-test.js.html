import reducerActiveItem from 'js/reducers/reducer_active_item';
import deepFreeze from 'deep-freeze';
 
test('returns the selected/active item', () => {
 
  const stateBefore =  {};
  const item = { title: 'Deus Ex-Machine', type: 'game'};
  const action = {
    type: 'ITEM_SELECTED',
    item
  };
  const stateAfter = item;
 
  deepFreeze(action);
  deepFreeze(stateBefore);
 
  expect(reducerActiveItem(stateBefore, action)).toEqual(stateAfter);
 
})
